[docs]def print_elapsed_time(viewfunc):
    @wraps(viewfunc)
    def inner(*args, **kw):
        t = time.perf_counter()
        res = viewfunc(*args, **kw)
        print("Done %s: %s sec" % (inner.__name__,
                                   Decimal(time.perf_counter() - t).quantize(
                                       Decimal("0.000000"))))
        return res

    return inner
